CREATE DATABASE CollegeDB;
USE CollegeDB;

1.Create table Student consisting of attributes student_id, name, mark, city, hod.

CREATE TABLE Student (
    std_id VARCHAR(10),
    name VARCHAR(20),
    mark INT,
    city VARCHAR(20),
    hod VARCHAR(20)
);

3.Create table Department with attributes dept_id, dname, hod and staff_no.

CREATE TABLE Department (
    dept_id VARCHAR(10) PRIMARY KEY,
    dname VARCHAR(20),
    hod VARCHAR(20),

);
